排序算法：简单选择，插入，冒泡

发表于2023-09-24|更新于2023-09-25|算法

简单选择排序

void SelectSort(int arr[], int size)
{
    int i, j;
    for (i = 0; i < size-1; i++)
    {
        int index = i;
        for(j = i+1; j < size; j++)
        {
            if(arr[index] > arr[j])
            {
                index = j;
            }
        }
        int temp = arr[i];
        arr[i] = arr[index];    
        arr[index] = temp;
    }
}

插入排序

void InsertSort(int arr[], int size)
{
    int i, j;
    for(i = 1; i < size; i++)
    {
        int temp = arr[i]; 
        j=i-1;
        while (j >= 0 && arr[j] > temp)
        {
            arr[j+1] = arr[j];
            j–;
        }
        arr[j+1] = temp;
    }
}
